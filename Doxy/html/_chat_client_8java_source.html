158 protected String hashPassword(String password) {
159 try {
160 MessageDigest digest = MessageDigest.getInstance("SHA-256");
161 byte[] encodedHash = digest.digest(password.getBytes(StandardCharsets.UTF_8));
162 StringBuilder hexString = new StringBuilder();
163 for (byte b : encodedHash) {
164 hexString.append(String.format("%02x", b));
165 }
166 return hexString.toString();
167 } catch (NoSuchAlgorithmException e) {
168 throw new RuntimeException("SHA-256 not supported");
169 }

186 private String hashPassword(String password) {
187 try {
188 MessageDigest digest = MessageDigest.getInstance("SHA-256");
189 byte[] encodedHash = digest.digest(password.getBytes(StandardCharsets.UTF_8));
190 StringBuilder hexString = new StringBuilder();
191 for (byte b : encodedHash) {
